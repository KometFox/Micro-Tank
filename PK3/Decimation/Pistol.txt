
ACTOR DeciPistol : Weapon 
{
   Weapon.AmmoUse 1
   Weapon.AmmoGive 20
   Weapon.AmmoType "Clip"
   Inventory.PickupMessage "$GOTCHAINGUN" // "You got the chaingun"
   Obituary "$OB_MPCHAINGUN" // "%o was mowed down by %k's chaingun."
   Tag "$TAG_CHAINGUN"
     Weapon.SlotNumber 2

   States
   {
   Ready:
     CHGG A 1 A_WeaponReady
     Loop
   Deselect:
     CHGG A 1 A_Lower
     Loop
   Select:
     CHGG A 1 A_Raise
     Loop
   Fire:
     CHGF AB 4 A_FireBullets(2.0,1.8,-1,15, "BulletPuff", FBF_USEAMMO|FBF_NORANDOM)
     CHGF B 0 A_ReFire
     Goto Ready
   Flash:
     CHGG A 5 Bright A_Light1
     Goto LightDone
     CHGG B 5 Bright A_Light1
     Goto LightDone
   Spawn:
     MGUN A -1
     Stop
   }
}
